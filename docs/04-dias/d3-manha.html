Ao final deste capítulo, você será capaz de:

  • Manipular texto com stringr (buscar, extrair, substituir)

Manipulação de texto com stringr

Dados reais vêm sujos: espaços extras, maiúsculas inconsistentes, padrões que precisam ser extraídos. O pacote stringr resolve isso.

Detectar e contar padrões

library(stringr)

# Quantos filmes têm "The" no título?
filmes %>%
  mutate(tem_the = str_detect(titulo, "^The ")) %>%
  count(tem_the)
# A tibble: 2 × 2
  tem_the     n
  <lgl>   <int>
1 FALSE    3082
2 TRUE      793
# Quantas espécies contêm "Eucalyptus"?
inv_florestal %>%
  mutate(euca = str_detect(especie, "Eucalyptus")) %>%
  count(euca)
# A tibble: 2 × 2
  euca      n
  <lgl> <int>
1 FALSE   350
2 TRUE    781

Extrair e substituir

# Extrair o ano de uma data em formato texto
str_extract("15/03/2023", "\\d{4}$")
[1] "2023"
# Substituir abreviações
str_replace("Eucalyptus grandis", "Eucalyptus", "E.")
[1] "E. grandis"
# Padronizar nomes de espécies
inv_florestal %>%
  mutate(especie_curto = str_replace(especie, "Eucalyptus", "E.")) %>%
  distinct(especie_curto)
# A tibble: 3 × 1
  especie_curto
  <chr>        
1 E. Grandis   
2 Pinus Taeda  
3 E. Urophylla

Limpeza de texto

# Texto sujo (simulando dado real de campo)
nomes_sujos <- c("  EUCALYPTUS GRANDIS  ", "Pinus taeda", "Eucalyptus  urophylla")

# Faxina completa
nomes_sujos %>%
  str_squish() %>%        # remove espaços extras
  str_to_lower() %>%      # tudo minúsculo
  str_to_title()          # primeira letra maiúscula
[1] "Eucalyptus Grandis"   "Pinus Taeda"          "Eucalyptus Urophylla"
# Aplicando limpeza ao dataset
inv_florestal %>%
  mutate(especie_limpa = str_squish(especie) %>% str_to_title()) %>%
  count(especie_limpa)
# A tibble: 3 × 2
  especie_limpa            n
  <chr>                <int>
1 Eucalyptus Grandis     682
2 Eucalyptus Urophylla    99
3 Pinus Taeda            350

Fatores com forcats

Fatores são variáveis categóricas. O pacote forcats ajuda a reordená-los e agrupá-los — essencial para gráficos e modelos.

Reordenar por outra variável

# Ordenar espécies por DAP médio (útil para gráficos)
inv_florestal %>%
  mutate(especie = fct_reorder(especie, dap, .fun = median)) %>%
  ggplot(aes(x = especie, y = dap, fill = especie)) +
  geom_boxplot() +
  labs(x = NULL, y = "DAP (cm)", title = "DAP por espécie (ordenado pela mediana)") +
  theme(legend.position = "none")

Agrupar categorias raras

# Agrupar gêneros com poucos filmes como "Outros"
filmes %>%
  mutate(genero_agrupado = fct_lump(genero, n = 5)) %>%
  count(genero_agrupado, sort = TRUE)
# A tibble: 6 × 2
  genero_agrupado       n
  <fct>             <int>
1 Other              1521
2 Drama               719
3 Ação                566
4 Comédia             452
5 Terror              311
6 Ficção científica   306
# Agrupar espécies raras no inventário
inv_florestal %>%
  mutate(especie_agrupada = fct_lump(especie, n = 3)) %>%
  count(especie_agrupada, sort = TRUE)
# A tibble: 3 × 2
  especie_agrupada         n
  <fct>                <int>
1 Eucalyptus Grandis     682
2 Pinus Taeda            350
3 Eucalyptus Urophylla    99

Ordenar por frequência

# Gênero mais comum primeiro
filmes %>%
  mutate(genero = fct_infreq(genero)) %>%
  ggplot(aes(y = genero)) +
  geom_bar(fill = "#18BC9C") +
  labs(title = "Frequência de gêneros", x = "Nº de filmes", y = NULL)

Datas com lubridate

Lidar com datas no R sempre foi doloroso — até o lubridate. Ele transforma texto em data com funções intuitivas.

Parse de datas

library(lubridate)

# Formatos comuns no Brasil
dmy("15/03/2023")       # dia/mês/ano
[1] "2023-03-15"
dmy("15 de março de 2023")
[1] "2023-03-15"
ymd("2023-03-15")       # ano-mês-dia (ISO)
[1] "2023-03-15"
mdy("March 15, 2023")   # mês/dia/ano (inglês)
[1] "2023-03-15"
# Com hora
ymd_hms("2023-03-15 14:30:00")
[1] "2023-03-15 14:30:00 UTC"
dmy_hm("15/03/2023 14:30")
[1] "2023-03-15 14:30:00 UTC"

Extrair componentes

data_medicao <- dmy("15/03/2023")

year(data_medicao)     # ano
[1] 2023
month(data_medicao)    # mês (número)
[1] 3
month(data_medicao, label = TRUE)  # mês (nome)
[1] Mar
12 Levels: Jan < Feb < Mar < Apr < May < Jun < Jul < Aug < Sep < ... < Dec
wday(data_medicao, label = TRUE)   # dia da semana
[1] Wed
Levels: Sun < Mon < Tue < Wed < Thu < Fri < Sat
quarter(data_medicao)  # trimestre
[1] 1
semester(data_medicao) # semestre
[1] 1

Aritmética com datas

# Adicionar dias
data_medicao + days(30)
[1] "2023-04-14"
# Adicionar meses
data_medicao + months(6)
[1] "2023-09-15"
# Adicionar anos
data_medicao + years(1)
[1] "2024-03-15"
# Calcular idade do povoamento
data_plantio <- dmy("15/05/2017")
intervalo <- interval(data_plantio, data_medicao)
intervalo / years(1)    # idade em anos decimais
[1] 5.832877

Exemplo prático: dados climáticos

clima <- read_csv("../data/clima.csv")

clima %>%
  mutate(
    data = dmy(data),
    mes = month(data, label = TRUE, abbr = FALSE),
    ano = year(data)
  ) %>%
  group_by(mes) %>%
  summarise(
    temp_media = mean(temperatura, na.rm = TRUE),
    precip_total = sum(precipitacao, na.rm = TRUE)
  )
# A tibble: 12 × 3

 3 March           24.7         886.
 4 April           25.1         942.
 5 May             23.7         977.
 6 June            21.5         662.
 7 July            18.5         556.
 8 August          16.0         359.
 9 September       15.5         346.
10 October         15.2         305.
11 November        16.3         390.

# Série temporal de temperatura
clima %>%
  mutate(data = dmy(data)) %>%
  ggplot(aes(x = data, y = temperatura)) +
  geom_line(color = "#E74C3C") +
  geom_smooth(se = FALSE, color = "#2C3E50") +
  labs(title = "Temperatura ao longo do tempo", x = NULL, y = "Temperatura (°C)")
